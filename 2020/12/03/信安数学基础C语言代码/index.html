#include <stdio.h>
#include <math.h>
int getmod(int b,int n,int m);//模重复平方
int main() {

	int n,sign=1,num;
	int s,t,i,j,h,h1;
	int b;
	printf("信安实验191，2019122061，颜礼能\n");
	printf("输入要检验的数：");
	scanf("%d",&n);
	printf("求模使用的第一题的模重复平方计算法\n");
	if(n%2==0) {
		printf("%d是合数",n);
		return 0;
	} else {
		num=(n-1)/2;
		while(num%2==0) {
			sign++;
			num=num/2;
		}
	}
	s=sign;
	t=num;
	printf("s=%d,t=%d\n",s,t);
	for(i=2; i<sqrt(n); i++) {

		h=getmod(i,t,n);
		printf("\n当b=%d，",i);
		for(j=0; j<s; j++) {
			h1=pow(2,j);
			b=getmod(h,h1,n);
			printf("b^%d^%d=%d，",t,j,b);
			if(b == 1 || b == n-1) {
				printf("是素数\n");
				break;
			}
			if(j==s-1&&b!=n-1) {
				printf("是合数\n");
				return 0;
			}
			if(b==n-1) {
				printf("是素数\n");
				break;
			}

		}
	}
	printf("结论：%d是素数\n",n);
	return 0;
}
